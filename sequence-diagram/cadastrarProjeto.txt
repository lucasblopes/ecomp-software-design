sequenceDiagram


    Ctrl->>View: showCadastroProjetoScreen()
    Ctrl->>View: inputTitulo()
    View-->>Ctrl: titulo

    Ctrl->>Repository: buscarPorTitulo(titulo)
    Repository-->>Ctrl: Optional<Projeto>

    alt Título já existe
        Ctrl->>View: showError("Título duplicado")
    else Título único
        Ctrl->>View: inputCliente()
        Ctrl->>View: inputDataInicial()
        Ctrl->>View: inputPrazoEntrega()
        Ctrl->>View: inputDescricao()
        View-->>Ctrl: dados do projeto

        Ctrl->>Projeto: new Projeto(titulo, cliente, dataInicio, prazoEntrega, descricao)
        
        alt Projeto inválido
            Ctrl->>View: showError("Campos não preenchidos")
        else Projeto válido
            Ctrl->>View: inputCaminhoContratoPDF()
            View-->>Ctrl: contratoPDF

            alt Contrato existe
                Ctrl->>Projeto: setContratoPDF(contratoPDF)
            end

            Ctrl->>Repository: adicionarProjeto(projeto)
            Ctrl->>View: showSuccess("Projeto cadastrado")
        end
    end
